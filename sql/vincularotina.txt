create table AGENDA_VINCULARROTINA(
    CODROTINA integer,
    CODUSUARIO integer,
    PRIORITARIO char(1),
    REAGENDAVEL char(1),
    HORARIOFIXO char(1),
    PERIODO varchar(15),
    DIAS VARCHAR(255),
    MESES VARCHAR(255),
    ANOS VARCHAR(255),
    DIASDASEMANA VARCHAR(255),
    DIASUTEIS VARCHAR(255),
    HORARIOS VARCHAR(255),
    CONSTRAINT FK_ROTINA FOREIGN KEY (CODROTINA) REFERENCES AGENDA_ROTINA(CODROTINA),
    CONSTRAINT FK_USUARIO FOREIGN KEY (CODUSUARIO) REFERENCES USUARIOS(CODUSUARIO),
    CONSTRAINT PK_ROTINAUSUARIO PRIMARY KEY (CODROTINA, CODUSUARIO)
);
